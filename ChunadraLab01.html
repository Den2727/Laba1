using namespace std;

int main() {

setlocale(LC_ALL, "Ukrainian");

struct Pair {

double first;
double second;

void Init(double f, double s) {

first = f;
second = s;
}

void Read() {

cout << "Введіть значення для first: ";
cin >> first;
cout << "Введіть значення для second: ";
cin >> second;
}

void Display() {

cout << "Перше число (first): " << first << endl;
cout << "Друге число (second): " << second << endl;
}

double power() {

return pow(first, second);
}
};

Pair myPair;

myPair.Init(0.0, 0.0);

myPair.Read();

myPair.Display();

double result = myPair.power();
